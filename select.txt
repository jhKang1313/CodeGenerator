SELECT
	C.CD_MODULE,
	CD_FIELD,
	NM_FIELD,
	NM2_FIELD,
	YN_SYSCODE,
	ID_INSERT,
	DTS_INSERT,
	ID_UPDATE,
	DTS_UPDATE

FROM 
	CI_CODE C
WHERE 
	CD_MODULE = {@P_CD_MODULE}
	AND CD_FIELD = {@P_CD_FIELD}
	AND (
		CD_FIELD LIKE '%' || {@P_KEYWORD} || '%'
		OR NM_FIELD LIKE '%' || {@P_KEYWORD} || '%'
		OR NM2_FIELD LIKE '%' || {@P_KEYWORD} || '%'
	)
